def monthly_sales(data):
    monthly_data = data.copy()
    monthly_data.fecha = monthly_data.fecha.apply(lambda x: str(x)[:-3])
    monthly_data = monthly_data.groupby('fecha')['ventas'].sum().reset_index()
    monthly_data.fecha = pd.to_datetime(monthly_data.fecha)
    return monthly_data

    number_of_years = number_of_days.days / 365

def time_plot(data, x_col, y_col, title):
    fig, ax = plt.subplots(figsize=(15,5))
    sns.lineplot(x_col, y_col, data=data, ax=ax, color='#666', label='Ventas totales')
    
    second = data.groupby(data.fecha.dt.year)[y_col].mean().reset_index()
    second.fecha = pd.to_datetime(second.fecha, format='%Y')
    sns.lineplot((second.fecha + datetime.timedelta(6*365/12)), y_col, data=second, ax=ax, color='#ff6961', label='Media de ventas')   
    
    ax.set(xlabel = "Fecha",
           ylabel = "Ventas",
           title = title)
    
    sns.despine()

def plots(data, lags=None):

    dt_data = data.set_index('fecha').drop('ventas', axis=1)
    dt_data.dropna(axis=0)
    
    layout = (1, 3)
    raw  = plt.subplot2grid(layout, (0, 0))
    acf  = plt.subplot2grid(layout, (0, 1))
    pacf = plt.subplot2grid(layout, (0, 2))
    
    dt_data.plot(ax=raw, figsize=(12, 5), color='#666')
    smt.graphics.plot_acf(dt_data, lags=lags, ax=acf, color='#666')
    smt.graphics.plot_pacf(dt_data, lags=lags, ax=pacf, color='#666')
    sns.despine()
    plt.tight_layout()

#Creamos un dataframe para transformar de series de tiempo a supervisadas 
def generate_supervised(data):
    supervised_df = data.copy()
    
    #Creamos una columna por cada lag
    for i in range(1,13):
        col_name = 'lag_' + str(i)
        supervised_df[col_name] = supervised_df['sales_diff'].shift(i)
    
    #Eliminar valores nulos
    supervised_df = supervised_df.dropna().reset_index(drop=True)
    
    supervised_df.to_csv('../data/model_df.csv', index=False)
    
    return supervised_df
